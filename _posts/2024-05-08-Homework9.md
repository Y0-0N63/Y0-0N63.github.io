# 6주차 스터디 문제 풀이

## 간단한 게시판 만들기

```
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>게시판 만들기</title>
    <style>
      body {
        font-family: Arial, Helvetica, sans-serif;
      }

      table {
        border-collapse: collapse;
        width: 1000px;
      }

      button {
        width: 150px;
        height: 40px;
        font-size: 16px;
        border-radius: 15px;
        margin-bottom: 10px;
      }

      input {
        height: 22px;
        margin-right: 5px;
      }
      #inputWriter {
        width: 250px;
      }

      #inputContent {
        width: 500px;
      }

      #writeBtn {
        width: 70px;
        height: 30px;
        font-size: 15px;
        border-radius: 12px;
      }

      th,
      td {
        border: 1px solid black;
        padding: 2px;
      }

      #numHead {
        width: 5%;
      }

      #writerHead {
        width: 10%;
      }

      #contentHead {
        width: 45%;
      }

      #dateHead {
        width: 25%;
      }

      #funcHead {
        width: 15%;
      }
    </style>
  </head>
  <body>
    <h1>게시판 만들기</h1>
    <button onclick="downloadText('posts.json')">저장하기</button>
    <button for="fileBtn" class="function-btn" id="open_file">불러오기</button>
    <input
      type="file"
      id="fileBtn"
      style="display: none"
      accept=".json"
      title="Select JSON File"
    />
    <button onclick="deleteAllPosts()">게시판 비우기</button>
    <br />
    <input
      type="text"
      id="inputWriter"
      placeholder="작성자를 입력해주세요."
      onkeypress="if(event.keyCode==13)document.querySelector('#inputContent').focus()"
    />
    <input
      type="text"
      id="inputContent"
      placeholder="글 내용을 입력해주세요."
      onkeypress="if(event.keyCode==13) {addPost(); document.querySelector('#inputWriter').focus();}"
    />
    <button
      onclick="{addPost(); document.querySelector('#inputWriter').focus();}"
      id="writeBtn"
    >
      글 작성
    </button>
    <table id="postTable">
      <thead>
        <tr>
          <th id="numHead">번호</th>
          <th id="writerHead">작성자</th>
          <th id="contentHead">내용</th>
          <th id="dateHead">작성일</th>
          <th id="funcHead">기능</th>
        </tr>
      </thead>
      <tbody id="postList"></tbody>
    </table>

    <script>
      let posts = [];
      let postNum = 1;

      function addPost() {
        const writer = document.querySelector('#inputWriter').value;
        const content = document.querySelector('#inputContent').value;
        const currentTime = new Date().toLocaleString();

        if (writer.trim() === '' || content.trim() === '') {
          alert('작성자 혹은 글 내용을 입력해주세요.');
          return;
        }

        const post = {
          num: postNum++,
          writer: writer,
          content: content,
          time: currentTime,
        };

        posts.push(post);

        document.querySelector('#inputWriter').value = '';
        document.querySelector('#inputContent').value = '';

        updatePost();
      }

      function updatePost() {
        const postList = document.querySelector('#postList');
        postList.innerHTML = '';

        posts.forEach((post) => {
          const postRow = document.createElement('tr');

          postRow.innerHTML = `
          <td>${post.num}</td>
          <td>${post.writer}</td>
          <td>
            ${post.content}
            <br>
              ${post.comments ? '|' + post.comments.join('<br>| ') : ''}
          </td>
          <td>${post.time}</td>
          <td>
            <button onclick="deletePost(${post.num})">삭제</button>
            <button onclick="addComment(${post.num})">댓글</button>
          </td>
        `;

          postList.appendChild(postRow);
        });
      }

      function deletePost(postNum) {
        const confirmed = confirm('정말 삭제하시겠습니까?');
        if (confirmed) {
          posts = posts.filter((post) => post.num !== postNum);
          updatePost();
        }
      }

      function addComment(postNum) {
        const comment = prompt('댓글을 입력하세요.');
        if (comment !== null && comment.trim() !== '') {
          const post = posts.find((post) => post.num === postNum);
          post.comments = post.comments || [];
          const currentDate = new Date().toLocaleString();
          const commentWithDate = `${comment} (${currentDate})`;
          post.comments.push(commentWithDate);
          updatePost();
        }
      }

      function deleteAllPosts() {
        const deleteConfirm = confirm('정말 삭제하시겠습니까?');
        if (deleteConfirm) {
          postNum = 1;
          posts = [];
          updatePost();
        }
      }

      function downloadText(filename) {
        if (posts.length == 0) {
          alert('저장할 데이터가 없습니다.');
          return;
        }

        var element = document.createElement('a');
        var jsonString = JSON.stringify(posts, null, 4);

        element.setAttribute(
          'href',
          'data:text/plain;charset=utf-8,' + encodeURIComponent(jsonString)
        );
        element.setAttribute('download', filename);
        element.style.display = 'none';
        document.body.appendChild(element);
        element.click();

        document.body.removeChild(element);
      }

      const label = document.getElementById('open_file');
      const fileInput = document.getElementById('fileBtn');
      label.addEventListener('click', function (e) {
        e.preventDefault();
        fileInput.click();
      });

      fileInput.addEventListener('change', function (e) {
        const file = e.target.files[0];
        const fr = new FileReader();
        fr.onload = () => {
          var fileContent = fr.result;
          var jsonObj = JSON.parse(fileContent);
          posts = jsonObj;
          updatePost();
        };
        fr.readAsText(file);
      });
    </script>
  </body>
</html>
```
