## 0. 계산기 기능 추가하기

연산 기호를 눌렀을 때, 숫자를 지우고 입력한 기호만 띄워주도록 수정하였다.
(연산자/피연산자 구분)
'eval( )' 대신 'calculateResult( )'라는 함수를 만들어서 연산식을 직접 계산하도록 수정하였다.
입력한 연산식의 진행 상태를 볼 수 있도록 입력창 위에 작게 띄워주는 기능을 추가하였다.
계산 중 오류가 발생하면 'Error'를 띄워주는 기능을 추가하였다.

```

      var expression = '';

      var displayExpression = document.querySelector('#displayExpression');
      var displayResult = document.querySelector('#displayResult');

      // 숫자 눌렀을 때
      function displayNum(num) {
        // 새 버튼을 눌렀을 때, 입력창에 이전 연산의 결과가 남아있으면 결과 지워주기
        if (resultDisplayed) {
          resultDisplayed = false;
          expression = '';
          displayResult.textContent = '';
          displayExpression.textContent = '';
        }

        if (
          displayResult.textContent === '+' ||
          displayResult.textContent === '-' ||
          displayResult.textContent === '*' ||
          displayResult.textContent === '/'
        ) {
          // 입력창에 연산 부호가 남아있으면, 연산 부호를 지워주고(입력창을 공백으로) ✔️
          // 입력한 숫자만 입력창에 남겨주는 기능 추가 ✔️
          expression += num;
          displayResult.textContent = num;
          displayExpression.textContent = expression;
        } else {
          expression += num;
          displayResult.textContent += num;
          displayExpression.textContent = expression;
        }
      }

      // 연산 기호 눌렀을 때
      // 기존에 입력된 숫자 지우고 연산 기호만 띄워주는 기능 추가 ✔️
      function displaySymbol(symbol) {
        displayResult.textContent = '';
        expression += symbol;
        displayResult.textContent += symbol;
        displayExpression.textContent = expression;
      }

      // '=' 눌렀을 때
      function equal() {
        // 수식 계산하는 함수 호출
        var result = calculateResult(expression);
        displayExpression.textContent += '=';
        // 계산 후 결과 출력
        displayResult.textContent = result;
        // 결과 계산 후 연산식 초기화
        expression = '';
        resultDisplayed = true;
      }

      // 입력한 수식 계산 및 결과 반환
      function calculateResult(expression) {
        try {
          return Function('return ' + expression)();
          // 예외 발생->Error 반환
        } catch (error) {
          return 'Error';
        }
      }

      // C 버튼을 눌렀을 때
      function init() {
        expression = '';
        displayResult.textContent = '';
        displayExpression.textContent = '';
        resultDisplayed = false;
      }
    </script>
```

## 01. 오늘의 할 일 리스트 만들기

내가 만든 문제 😊😊
저번 주에 만들었던 코드에서 입력란에 문자열을 입력한 후, 엔터키를 눌러도 등록이 가능하도록 기능을 추가하였다.

```
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>투두 리스트 만들기</title>
    <style>
      .container {
        width: 400px;
        height: 500px;
        background-color: #dbddf9;
        padding: 10px;
        text-align: center;
      }

      input {
        text-align: center;
        width: 250px;
        height: 30px;
        font-size: 17px;
      }

      button {
        width: 50px;
        height: 33px;
      }

      #list {
        text-align: left;
        font-size: 17px;
      }
    </style>
  </head>
  <body>
    <div class="container">
      <h1>오늘의 할 일</h1>
      <!-- 엔터 키를 눌렀을 때 리스트에 할 일 추가시켜주기-->
      <input
        type="text"
        id="inputList"
        placeholder="오늘의 할 일을 입력하세요"
        onkeypress="if(event.keyCode==13) addList();"
      />
      <!-- onkeypress() : 키보드로 문자 입력할 때 발생
      키 코드가 13=엔터키 -->

      <button onclick="addList()">추가</button>
      <ul id="list"></ul>
    </div>

    <script>
      const toDoList = document.querySelector('#list');

      function addList() {
        const input = document.querySelector('#inputList');
        const toDo = input.value.trim();

        if (toDo !== '') {
          const toDoItem = document.createElement('ul');
          toDoItem.textContent = '◽ ' + toDo;
          toDoList.appendChild(toDoItem);
          input.value = '';
        } else {
          alert('할 일을 입력하세요');
        }
      }
    </script>
  </body>
</html>
```
