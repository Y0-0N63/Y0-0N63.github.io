# 회원 가입 화면 만들기

```
<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>멤버쉽 회원 가입 관리</title>
    <style>
      fieldset {
        height: 100%;
        width: 50%;
      }

      .container {
        display: flex;
        flex-direction: column;
        flex-wrap: nowrap;
        justify-content: flex-end;
      }

      .row {
        margin-bottom: 2px;
      }

      label {
        display: inline-block;
        width: 140px;
      }
    </style>
  </head>
  <body>
    <form>
      <fieldset>
        <legend>회원가입</legend>
        <div class="container">
          <div class="row">
            <label for="id">아이디</label>
            <input
              class="id"
              type="text"
              id="id"
              required="required"
              placeholder="아이디를 입력하세요"
              size="25"
            />
          </div>

          <div class="row">
            <label for="pwd">비밀번호</label>
            <input
              type="password"
              id="pwd"
              placeholder="비밀번호를 입력하세요"
              oninput="PwdCheck()"
              size="25"
            />
          </div>

          <div class="row">
            <label for="repwd">비밀번호 재확인</label>
            <input
              type="password"
              id="repwd"
              placeholder="비밀번호를 다시 입력하세요"
              oninput="PwdCheck()"
              size="25"
            />
          </div>

          &nbsp;

          <div class="row name">
            <label for="name">이름</label>
            <input type="text" id="name" />
          </div>

          <div class="row gender">
            <label>성별</label>
            <input type="radio" name="gender" value="man" checked />남자
            <input type="radio" name="gender" value="woman" />여자
          </div>

          <div class="row birthday">
            <label for="birthdate">생일</label>
            <input type="date" id="birthdate" name="date" />
          </div>

          <div class="row email">
            <label for="email">본인 확인(이메일)</label>
            <input
              type="email"
              name="email"
              size="25"
              placeholder="본인 확인(이메일)"
            />
          </div>

          &nbsp;

          <div class="row phone">
            <select>
              <option>+82</option>
            </select>
            <input type="text" id="phone" size="15" />
            <input type="button" value="인증" id="checkphone" />
          </div>

          <div class="row num">
            <input
              type="text"
              id="verificationCode"
              size="22"
              placeholder="인증번호를 입력하세요"
            />
            <input type="button" value="확인" id="confirm" />
          </div>

          <!-- 가입 버튼 추가 -->
          <div class="row">
            <input
              type="button"
              value="가입"
              onclick="goToLoginPage()"
              id="signupButton"
            />
          </div>
        </div>
      </fieldset>
    </form>
    <script>
      let pw = document.querySelector('#pwd');
      let pw2 = document.querySelector('#repwd');

      pw2.addEventListener('keyup', function () {
        if (pw.value !== pw2.value) {
          pw2.style.color = '#ff0000';
        } else {
          pw2.style.color = '#000';
        }
      });

      let checkBtn = document.querySelector('#checkphone');
      checkBtn.addEventListener('click', function () {
        let phone = document.querySelector('#phone');
        if (phone.value === '1234') {
          alert('인증되었습니다.');
          phone.readOnly = true;
          phone.disabled = true;
        } else {
          phone.focus();
          alert('인증번호가 틀렸습니다.');
        }
      });

      let joinBtn = document.querySelector('#signupButton');
      joinBtn.addEventListener('click', function () {
        alert('회원가입이 완료되었습니다.');
        location.href = 'login.html';
      });
    </script>
  </body>
</html>
```
